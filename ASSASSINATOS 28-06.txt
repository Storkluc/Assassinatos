//O procedimento abaixo executa animações em forma de barra.

   Escreva ("100%")
   Escreval ("")
   Escreva  ("      Loading")
   Enquanto l[2]<>5 faca
      timer (100)
      Escreva ("_")
      l[2]<-1+l[2]
   fimenquanto
   Escreva("Error")
   Enquanto l[3]<>3 faca
      timer (100)
      Escreva ("_")
      l[3]<-1+l[3]
   fimenquanto
   Escreva("Error")
   Enquanto l[4]<>5 faca
      timer (100)
      Escreva ("_")
      l[4]<-1+l[4]
   fimenquanto
   Escreva ("Comparing")
   Enquanto l[5]<>10 faca
      timer (100)
